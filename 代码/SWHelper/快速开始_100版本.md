### 示例2: 批量创建

```python
import win32com.client

helper = win32com.client.Dispatch("SWHelper.SWHelperDynamic")

helper.ConnectToSW()

# 创建多个不同尺寸的方块
sizes = [
    (50, 50, 25),
    (100, 100, 50),
    (150, 150, 75),
]

for width, depth, height in sizes:
    helper.CreatePart()
    helper.CreateSketch()
    helper.DrawRectangle(0, 0, width, depth)
    helper.CloseSketch()
    helper.SelectSketch("Sketch1")
    helper.CreateExtrusion(height)
    print(f"✅ 创建{width}x{depth}x{height}mm的方块")

print("✅ 批量创建完成！")
```
