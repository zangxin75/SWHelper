## 💻 完整实现代码

### 步骤1: 添加DrawPolygon方法

```csharp
/// <summary>
/// 绘制正多边形
/// </summary>
/// <param name="centerX">中心X坐标</param>
/// <param name="centerY">中心Y坐标</param>
/// <param name="radius">外接圆半径</param>
/// <param name="sides">边数（6=六边形）</param>
/// <returns>成功返回true</returns>
public bool DrawPolygon(double centerX, double centerY, double radius, int sides)
{
    try
    {
        if (sketchMgr == null)
        {
            lastError = "草图管理器未初始化";
            return false;
        }

        // 计算多边形顶点
        double angleStep = 2 * Math.PI / sides;
        double[,] vertices = new double[sides, 2];

        for (int i = 0; i < sides; i++)
        {
            double angle = i * angleStep;
            vertices[i, 0] = centerX + radius * Math.Cos(angle);
            vertices[i, 1] = centerY + radius * Math.Sin(angle);
        }

        // 使用dynamic绘制多边形线条
        for (int i = 0; i < sides; i++)
        {
            int next = (i + 1) % sides;
            sketchMgr.CreateLine(
                vertices[i, 0], vertices[i, 1], 0,
                vertices[next, 0], vertices[next, 1], 0
            );
        }

        lastError = "";
        return true;
    }
    catch (Exception ex)
    {
        lastError = "绘制多边形错误: " + ex.Message;
        return false;
    }
}
```

### 步骤2: 添加CreateThread方法

```csharp
/// <summary>
/// 创建螺纹特征
/// </summary>
/// <param name="diameter">螺纹直径</param>
/// <param name="pitch">螺距</param>
/// <param name="length">螺纹长度</param>
/// <returns>成功返回true</returns>
public bool CreateThread(double diameter, double pitch, double length)
{
    try
    {
        if (featureMgr == null)
        {
            lastError = "特征管理器未初始化";
            return false;
        }

        // 使用dynamic调用螺纹创建
        // 注意：SolidWorks螺纹创建需要特定的API调用
        dynamic feature = featureMgr.FeatureCutThreading(
            true,    // directionOne
            false,   // flipDirection
            false,   // useDefaultOffset
            false,   // useFeatureScope
            false,   // useAutoSelect
            true,    // maintainTangentChain
            false,   // isThinFeature
            false,   // isDraftFeature
            0.0,     // draftAngle
            0.0,     // offset
            length,  // depth
            false,   // reverseDirection
            false,   // useCapPlane
            0.0,     // capPlaneOffset
            diameter, // threadDiameter
            pitch,   // pitch
            0.0,     // minorDiameter
            60.0,    // includedAngle
            false,   // isMetric
            false,   // isCustom
            0,       // clearanceMethod
            0.0      // clearance
        );

        if (feature == null)
        {
            lastError = "无法创建螺纹，直径: " + diameter;
            return false;
        }

        lastError = "";
        return true;
    }
    catch (Exception ex)
    {
        lastError = "创建螺纹错误: " + ex.Message;
        return false;
    }
}
```

## 🔩 M5螺丝设计完整脚本

### Python设计脚本

```python
# -*- coding: utf-8 -*-
"""
M5外六角螺丝设计脚本
使用SWHelper创建完整的M5螺丝
"""

import win32com.client
import math

def create_m5_screw():
    """创建M5外六角螺丝"""

    # 连接SolidWorks
    helper = win32com.client.Dispatch("SWHelper.SWHelperDynamic")

    print("=" * 70)
    print("M5外六角螺丝设计")
    print("=" * 70)
    print()

    # 连接SolidWorks
    print("[1/8] 连接SolidWorks...")
    if not helper.ConnectToSW():
        print("错误:", helper.GetLastError())
        return False

    # 创建零件
    print("[2/8] 创建零件...")
    if not helper.CreatePart():
        print("错误:", helper.GetLastError())
        return False

    # M5螺丝参数
    thread_diameter = 5.0      # M5螺纹直径
    head_size_across = 8.0     # 六角对边距离
    head_radius = head_size_across / math.sqrt(3)  # 外接圆半径
    head_height = 5.0          # 头部高度
    shank_length = 16.0        # 杆部长度
    shank_diameter = 5.0       # 杆部直径
    thread_pitch = 0.8         # M5螺距
    thread_length = 12.0       # 螺纹长度

    print(f"参数:")
    print(f"  螺纹直径: M{thread_diameter}")
    print(f"  六角对边: {head_size_across}mm")
    print(f"  头部高度: {head_height}mm")
    print(f"  杆部长度: {shank_length}mm")
    print(f"  螺纹长度: {thread_length}mm")
    print()

    # 步骤1: 创建六边形头部草图
    print("[3/8] 创建六边形头部草图...")
    if not helper.CreateSketch():
        print("错误:", helper.GetLastError())
        return False

    # 绘制六边形（如果DrawPolygon可用）
    if hasattr(helper, 'DrawPolygon'):
        helper.DrawPolygon(0, 0, head_radius, 6)
        print("  使用DrawPolygon绘制六边形")
    else:
        print("  注意: DrawPolygon方法未实现，需要扩展")
        # 手动绘制六边形线条
        for i in range(6):
            angle = i * 2 * math.pi / 6
            next_angle = (i + 1) * 2 * math.pi / 6
            x1 = head_radius * math.cos(angle)
            y1 = head_radius * math.sin(angle)
            x2 = head_radius * math.cos(next_angle)
            y2 = head_radius * math.sin(next_angle)

            if hasattr(helper, 'DrawLine'):
                helper.DrawLine(x1, y1, x2, y2)

    # 关闭草图
    if not helper.CloseSketch():
        print("错误:", helper.GetLastError())
        return False

    # 拉伸六边形头部
    print("[4/8] 拉伸六边形头部...")
    if not helper.SelectSketch("Sketch1"):
        print("错误:", helper.GetLastError())
        return False

    if not helper.CreateExtrusion(head_height):
        print("错误:", helper.GetLastError())
        return False

    print(f"  头部创建: {head_size_across}mm x {head_height}mm")

    # 步骤2: 创建杆部
    print("[5/8] 创建圆柱杆部...")
    if not helper.CreateSketch():
        print("错误:", helper.GetLastError())
        return False

    # 绘制圆形（如果DrawCircle可用）
    if hasattr(helper, 'DrawCircle'):
        helper.DrawCircle(0, 0, shank_diameter / 2)
        print("  使用DrawCircle绘制圆形")
    else:
        print("  注意: DrawCircle方法未实现，需要扩展")

    if not helper.CloseSketch():
        print("错误:", helper.GetLastError())
        return False

    # 拉伸杆部
    if not helper.SelectSketch("Sketch2"):
        print("错误:", helper.GetLastError())
        return False

    if not helper.CreateExtrusion(shank_length):
        print("错误:", helper.GetLastError())
        return False

    print(f"  杆部创建: Ø{shank_diameter}mm x {shank_length}mm")

    # 步骤3: 创建头部倒角
    print("[6/8] 创建头部倒角...")
    if hasattr(helper, 'CreateChamfer'):
        helper.CreateChamfer(0.5, 45.0)
        print("  头部倒角: 0.5mm x 45°")
    else:
        print("  注意: CreateChamfer方法未实现")

    # 步骤4: 创建杆部倒角
    print("[7/8] 创建杆部倒角...")
    if hasattr(helper, 'CreateChamfer'):
        helper.CreateChamfer(0.3, 45.0)
        print("  杆部倒角: 0.3mm x 45°")
    else:
        print("  注意: CreateChamfer方法未实现")

    # 步骤5: 创建螺纹
    print("[8/8] 创建螺纹...")
    if hasattr(helper, 'CreateThread'):
        helper.CreateThread(thread_diameter, thread_pitch, thread_length)
        print(f"  螺纹创建: M{thread_diameter}, 螺距{thread_pitch}mm, 长度{thread_length}mm")
    else:
        print("  注意: CreateThread方法未实现")

    print()
    print("=" * 70)
    print("M5螺丝设计完成！")
    print("=" * 70)
    print()
    print("设计总结:")
    print(f"  规格: M5 x {shank_length}")
    print(f"  头部: 六角 {head_size_across}mm x {head_height}mm")
    print(f"  杆部: Ø{shank_diameter}mm x {shank_length}mm")
    print(f"  螺纹: M{thread_diameter} x {thread_pitch}mm, 长度{thread_length}mm")
    print()

    return True

# 执行设计
if __name__ == "__main__":
    try:
        success = create_m5_screw()
        if success:
            print("✓ 设计成功！")
        else:
            print("✗ 设计失败，请检查错误信息")
    except Exception as e:
        print(f"✗ 发生错误: {e}")
        import traceback
        traceback.print_exc()
```
